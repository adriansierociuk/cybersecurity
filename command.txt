/* -- Nazwa komputera -- */
hostname
cat /etc/hostname 

/* -- Czas, strefa czasowa -- */
date 
cat /etc/timezone 
ls -la /etc/location

/* -- Dystrybucja, wersja -- */
lsb_relase -a
cat /etc/*-relase 
cat /etc/issue.net

/* -- Dysk, partycje -- */
fdisk -l
mount 
cat /etc/fstab

/* -- Zaplanowane zadania -- */
crontab -l
crontab -l -u $USER
ls -laR /etc/cron{tab,.*}
ls -laR /var/spool/cron

/* -- Polecenia uruchamiane przy starcie -- */
cat /etc/ec.local
ls -laR /etc/rc*.d 
find / -path "*/.config/autostart"
ls -laR /etc/init{,.d}

/* -- Polecenia uruchamiane przy logowaniu -- */
cat /etc/bash.bashrc
ls -laR /etc/profile{,.d}
find / -name .bashrc -o -name .bash_profile -o -name .profile

/* -- Sieć -- */
ifconfig
ip addr 
cat /etc/network/interfaces
egrep -v "^#\^$" /etc/dhcp/dhclient.conf 
cat /etc/hosts
cat /etc/resolv.conf
arp
route 
ip route 
netstat -plunt
lsof -i
tcpdump
netstat --unix
lsof -U

/* -- Zainstalowane oprogramowanie -- */
dpgk -l
grep ^Package: /var/lib/dpkg/status
ls -laR /var/cache/apt/archives
cat /var/log/dpkg.log
ls -la /usr/local/ /opt/
for dir in $(echo $PATH | sed 's/:/ /g'); do ls -laR $dir; done

/* -- Repozytorium pakietów -- */  
egrep -v "^#|^$' /etc/apt/sources.list
egrep -vh "^#|^$" /etc/apt/sources.list.d/*

/* -- Dostępne usługi -- */
ls -laR /etc/init{,.d}
ls -laR /etc/rx*.d
chkconfig --list 

/* Działające procesy */
ps auswww
ps auswwe
ps -elfH
lsof +L1


/* -- Informacje o rotowaniu się logów, jak długo są przechowywane stare logi, itp. -- */
cat /etc/logrotate.conf 
ls -laR /etc/logrotate.d
